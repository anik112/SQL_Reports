SELECT 
	   TB_MONTH_SALARY.CARDNO AS CARD_NO,
	   TB_MONTH_SALARY.EMPNAME AS EMP_NAME,
	   TB_MONTH_SALARY.DESIGNATION,
	   TB_MONTH_SALARY.SALARY_GRD AS GRADE,
	   TB_PERSONAL_INFO.JOINING_DATE AS JOINING_DATE,
	   TB_MONTH_TOTALDAY.TOTALPRESENT AS PRST_DAY,
	   TB_MONTH_TOTALDAY.TOTALLEAVE AS LEV_DAY,
	   TB_MONTH_SALARY.BASICSAL AS BASIC_SALARY,
	   TB_MONTH_SALARY.HOUSERENT AS HOUSE_RENT,
	   TB_MONTH_SALARY.MEDICAL AS MEDICAL,
	   TB_MONTH_SALARY.TRANSPORT AS CONVA,
	   TB_MONTH_SALARY.FOODALLOW AS F_ALLOW,
	   TB_MONTH_SALARY.GROSSSAL AS GROSS_SALARY,
	   TB_MONTH_TOTALDAY.TOTALABSENT AS ADSENT_DAY,
	   TB_MONTH_SALARY.ABSENT_DEDUC AS ABSENT_AMOUNT,
	   TB_MONTH_SALARY.ARREAR_AMT AS ARRER_AMOUNT,
	   TB_MONTH_TOTALDAY.LATEDAY AS LET,
	   TB_MONTH_TOTALDAY.ATTBONUS AS BONUS,
	   TB_MONTH_SALARY.TOTALSALARY AS TOTAL_WAGESS_PAYABLE,
	   TB_MONTH_TOTALDAY.OTHR AS OT_HRS,
	   TB_MONTH_SALARY.OTRATE AS OT_RATE,
	   TB_MONTH_SALARY.OTAMOUNT AS OT_AMOUNT,
	   TB_MONTH_SALARY.ADVANCE_DEDUC AS ADVANCE_DECUCTION,
	   TB_MONTH_SALARY.STAMP AS STAMP_DECUTION,
	   TB_MONTH_SALARY.NETPAYMENT AS NET_PAYMENT,
	   TB_MONTH_SALARY.FINMONTH AS SALARY_MONTH,
	   TB_MONTH_TOTALDAY.FINMONTH AS TOTAL_DAY_MONTH
	    
FROM 
	   TB_MONTH_SALARY,TB_MONTH_TOTALDAY,TB_PERSONAL_INFO
WHERE
	 TB_MONTH_SALARY.CARDNO='000152'
	 AND TB_MONTH_SALARY.CARDNO=TB_PERSONAL_INFO.CARDNO
	 AND TB_PERSONAL_INFO.CARDNO=TB_MONTH_TOTALDAY.CARDNO
	 AND TB_MONTH_SALARY.FINYEAR=2019
	 AND TB_MONTH_TOTALDAY.FINYEAR=2019
	 AND TB_MONTH_SALARY.FINMONTH='January'
	 AND TB_MONTH_TOTALDAY.FINMONTH='January'
	 AND TB_MONTH_SALARY.COMPANY='Landmark Footwear Ltd.'
	 AND TB_MONTH_TOTALDAY.COMPANY='Landmark Footwear Ltd.';
